2. KNN
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split
from sklearn.neighbors import KNeighborsClassifier

X, y = load_iris(return_X_y=True)
data = load_iris()
Xtr, Xte, ytr, yte = train_test_split(X, y, test_size=0.3, random_state=42)

p = KNeighborsClassifier(5).fit(Xtr, ytr).predict(Xte)

for i in range(len(yte)):
    status = "Correct" if yte[i] == p[i] else "Wrong"
    print(f"Actual: {data.target_names[yte[i]]:<15} Predicted: {data.target_names[p[i]]:<15} → {status}")

print(f"\nAccuracy: {sum(yte==p)/len(yte)*100:.2f}%")

from sklearn.datasets import load_iris

5. Linear Regression
 
import pandas as pd
import matplotlib.pyplot as plt
import numpy as np
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error, mean_absolute_error, r2_score

iris = load_iris()

X = iris.data[:, [0]]
y = iris.data[:, 2]

X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=1)

model = LinearRegression().fit(X_train, y_train)
y_pred = model.predict(X_test)

print(pd.DataFrame({
    "Actual": y_test.round(2),
    "Predicted": y_pred.round(2)
}))

print("\nValidation Table")
print(pd.DataFrame({
    "MAE": [round(mean_absolute_error(y_test, y_pred), 2)],
    "MSE": [round(mean_squared_error(y_test, y_pred), 2)],
    "RMSE": [round(np.sqrt(mean_squared_error(y_test, y_pred)), 2)],
    "R2": [round(r2_score(y_test, y_pred), 2)]
}))

plt.scatter(X_test, y_test, label="Actual Data")
plt.plot(X_test, y_pred, "r", label="Regression Line")
plt.legend()
plt.show()








6. Back Propagation (MLP)
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split
from sklearn.neural_network import MLPClassifier
from sklearn.metrics import accuracy_score, precision_score, recall_score, f1_score

iris = load_iris()
X, y = iris.data, iris.target
names = iris.target_names

X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.3, random_state=1
)

model = MLPClassifier(hidden_layer_sizes=(10,), max_iter=500, random_state=1)
model.fit(X_train, y_train)

y_pred = model.predict(X_test)

result = pd.DataFrame({
    "Actual": [names[i] for i in y_test],
    "Predicted": [names[i] for i in y_pred]
})

result["Result"] = ["Correct" if a == p else "Wrong"
                    for a, p in zip(y_test, y_pred)]

print(result)

print("\nCorrect Predictions =", (y_test == y_pred).sum())
print("Wrong Predictions =", (y_test != y_pred).sum())

print("\nValidation Table")
print(pd.DataFrame({
    "Accuracy": [round(accuracy_score(y_test, y_pred), 2)],
    "Precision": [round(precision_score(y_test, y_pred, average="macro"), 2)],
    "Recall": [round(recall_score(y_test, y_pred, average="macro"), 2)],
    "F1-Score": [round(f1_score(y_test, y_pred, average="macro"), 2)]
}))

plt.plot(model.loss_curve_)
plt.title("Back Propagation Loss Curve")
plt.xlabel("Iterations")
plt.ylabel("Loss")
plt.show()


















7. K-Means
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split
from sklearn.cluster import KMeans
from scipy.stats import mode
from sklearn.metrics import accuracy_score
import numpy as np
import matplotlib.pyplot as plt

X, y = load_iris(return_X_y=True)
Xtr, Xte, ytr, yte = train_test_split(X, y, test_size=0.3, random_state=42)

km = KMeans(3, random_state=42, n_init=10).fit(Xtr)
c = km.predict(Xte)

l = np.zeros_like(c)
for i in range(3):
    l[c==i] = mode(yte[c==i], keepdims=True)[0]

print(f"Accuracy: {accuracy_score(yte, l)*100:.2f}%")

plt.scatter(Xte[:,0], Xte[:,1], c=c, cmap='viridis')
plt.scatter(km.cluster_centers_[:,0], km.cluster_centers_[:,1], s=200, c='red', marker='X')
plt.title("K-Means (k=3)"); plt.xlabel("Sepal Length"); plt.ylabel("Sepal Width")
plt.show()
